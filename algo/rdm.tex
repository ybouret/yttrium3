\section{Description}

Let's assume we have a cluster of $M$ active species and $N$ equilibria:
\begin{equation}
\forall i, E_i : \sum_j \nu_{ij} A_j = 0 : K_i=\prod_j C_j^{\nu_{ij}}
\end{equation}
We define the topology matrix:
\begin{equation}
\bm{\nu} = \begin{bmatrix} \nu_{ij} \end{bmatrix} \in \mathcal{M}_{N,M}
\end{equation}
Starting from $\vec C_0\in \mathbb{R}^M$, the reachable state by the extent vector $\vec \xi \in \mathbb{R}^N$ is:
\begin{equation}
	\vec C = \vec C_0 + \bm{\nu}^T \vec \xi 
\end{equation}


\section{Conservation Laws}

\subsection{Computation}
A conservation law is non-trivial, positive linear combination of species that is not affected by an extent.
They appear in the {\bf CoVen} algorithm of $\bm{\nu}^\perp$. If a law is found, then the involved species are conserved.
 
\subsection{Algebraic Form}
Each contraint is a row vector $\vec \Gamma_i$ of:
\begin{equation}
\bm{\Gamma} \in \mathcal{M}^+_{Nc,M},\;\;\mathrm{rank}(\bm \Gamma) = r_g \leq N_c
\end{equation}
Any modification orthogonal to $\bm \nu$ from $\vec C_0$ is:
\begin{equation}
\label{eq:gamma}
\vec{C} = \vec C_0 + \bm{\Gamma}^T \vec\varphi,\;\; \vec \varphi \in \mathbb{R}^{N_c}
\end{equation}

\subsection{Legality}

\begin{itemize}
\item
A concentration set $\vec C_0$ is legal iff:
\begin{equation}
\forall i, \langle \vec \Gamma _i \vert \vec C_0 \rangle \geq 0
\end{equation}
with $\vec \Gamma _i \in \mathbb{R}_+^M$ with at least two coefficients.

\item
Let us assume that:
\begin{equation}
	\langle \vec \Gamma _i \vert \vec C_0 \rangle  = - x_i,\; x_i>0
\end{equation}
 
\item We need to inject $\lambda \vec \Gamma _i$ such that:
\begin{equation}
	\vec C = \vec C_0 + \lambda \vec \Gamma _i \implies \langle \vec \Gamma _i \vert \vec C \rangle = 0 \implies \lambda = \dfrac{x_i}{\vec \Gamma_i^2}
\end{equation}

\item The injected concentration is:
\begin{equation}
	\delta \vec C = \dfrac{x_i}{\vec \Gamma_i^2} \vec \Gamma_i = - \dfrac{\langle \vec \Gamma _i \vert \vec C_0 \rangle}{\vec \Gamma_i^2}\vec \Gamma_i 
\end{equation} 

The projected state is directly:
\begin{equation}
	\vec C = \dfrac{1}{\vec \Gamma_i^2}\left[ \vec \Gamma_i^2 \bm{I} -  \vert \vec \Gamma _i \rangle \langle \vec \Gamma _i \vert\right] \vec C_0
\end{equation}

\item We note that:
\begin{equation}
	\delta \vec C ^2 = \dfrac{x_i^2}{\vec \Gamma_i^2} \implies  \vert \delta \vec C  \vert = \dfrac{x_i}{ \vert \vec \Gamma_i \vert}
\end{equation}
This allows to select the smallest injected concentration.
\end{itemize}
